const HeroesFilters = () => {

    const dispatch = useDispatch();
    const {request} = useHttp();

    // –ó–∞–ø—Ä–æ—Å –Ω–∞ —Å–µ—Ä–≤–µ—Ä –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è —Ñ–∏–ª—å—Ç—Ä–æ–≤ –∏ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ–π —Å–º–µ–Ω—ã —Å–æ—Å—Ç–æ—è–Ω–∏—è
